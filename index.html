<script type="module">

import { initializeApp } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-app.js";

import { getAuth, createUserWithEmailAndPassword, signInWithEmailAndPassword, signOut, onAuthStateChanged, GoogleAuthProvider, signInWithPopup }

from "https://www.gstatic.com/firebasejs/10.12.0/firebase-auth.js";

import { getFirestore, collection, addDoc, getDocs, deleteDoc, doc }

from "https://www.gstatic.com/firebasejs/10.12.0/firebase-firestore.js";


// 🔥 REEMPLAZA ESTO CON TU CONFIG DE FIREBASE

const firebaseConfig = {

  apiKey: "AIzaSyAnCT6glmf4n1zxzHJoHx4Z9l7yR0jju1I",

  authDomain: "calc-gong-fu.firebaseapp.com",

  projectId: "calc-gong-fu",

  storageBucket: "calc-gong-fu.firebasestorage.app",

  messagingSenderId: "112038141907",

  appId: "1:112038141907:web:dcffd7d691db0acb4e1926"

};


const app = initializeApp(firebaseConfig);

const auth = getAuth(app);

const db = getFirestore(app);

const provider = new GoogleAuthProvider();


let usuarioActual = null;


onAuthStateChanged(auth,(user)=>{

  if(user){

    usuarioActual=user;

    authSection.classList.add("hidden");

    appSection.classList.remove("hidden");

    userEmail.innerText=user.email;

    cargarTeteras();

  }else{

    authSection.classList.remove("hidden");

    appSection.classList.add("hidden");

  }

});


window.registrar=()=>createUserWithEmailAndPassword(auth,emailInput.value,passwordInput.value);

window.login=()=>signInWithEmailAndPassword(auth,emailInput.value,passwordInput.value);

window.loginGoogle=()=>signInWithPopup(auth,provider);

window.logout=()=>signOut(auth);


window.guardarTetera=async()=>{

  if(!nombreTetera.value||!volumenTetera.value) return;

  await addDoc(collection(db,"usuarios",usuarioActual.uid,"teteras"),{

    nombre:nombreTetera.value,

    volumen:parseFloat(volumenTetera.value)

  });

  nombreTetera.value="";

  cargarTeteras();

};


async function cargarTeteras(){

  listaTeteras.innerHTML="";

  const snapshot=await getDocs(collection(db,"usuarios",usuarioActual.uid,"teteras"));

  snapshot.forEach(d=>{

    const data=d.data();

    const div=document.createElement("div");

    div.className="tetera-item";

    div.innerHTML=`

      <span onclick="volumenTetera.value=${data.volumen};calcular()">

      ${data.nombre} (${data.volumen}ml)

      </span>

      <button onclick="eliminarTetera('${d.id}')">🗑</button>

    `;

    listaTeteras.appendChild(div);

  });

}


window.eliminarTetera=async(id)=>{

  await deleteDoc(doc(db,"usuarios",usuarioActual.uid,"teteras",id));

  cargarTeteras();

};


window.calcular=()=>{

  const volumen=parseFloat(volumenTetera.value);

  const ratio=parseFloat(tipoTe.value);

  const tipoTexto=tipoTe.options[tipoTe.selectedIndex].text;


  const temperaturas={

    "Blanco":"85-95°C",

    "Verde":"85°C",

    "Oolong":"100°C",

    "Pu erh Sheng":"95°C",

    "Pu erh Shou":"100°C",

    "Negro":"90°C"

  };


  const ratioMl=ratio/110;

  const desplazamiento=3.2;

  const gramos=(volumen*ratioMl)/(1+ratioMl*desplazamiento);

  const agua=volumen-(gramos*desplazamiento);


  gramosResultado.innerText=gramos.toFixed(2);

  aguaResultado.innerText=agua.toFixed(2);

  temperaturaResultado.innerText=temperaturas[tipoTexto]||"-";

};


window.toggleDark=()=>{

  document.body.classList.toggle("dark");

  localStorage.setItem("darkMode",document.body.classList.contains("dark"));

};


if(localStorage.getItem("darkMode")==="true"){

  document.body.classList.add("dark");

}


if("serviceWorker" in navigator){

  navigator.serviceWorker.register("service-worker.js");

}


calcular();
